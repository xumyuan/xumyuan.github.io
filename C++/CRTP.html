奇异递归模板模式 ​

奇异递归模板模式（Curiously Recurring Template Pattern）大致用法如下：

C++
template<class Z>
class Y {};
class X : public Y<X> {};

主要是为了实现编译时多态，感觉主要是为了减少一个虚函数的开销，属于一种性能压榨到极致的用法吧，大多数情况为了代码可读性更高，还是可以使用虚函数继承多态来实现类似的功能

一个具体的使用示例

C++
#include <cstdio>
 
#ifndef __cpp_explicit_this_parameter // Traditional syntax
 
template <class Derived>
struct Base
{
    void name() { static_cast<Derived*>(this)->impl(); }
protected:
    Base() = default; // prohibits the creation of Base objects, which is UB
};
struct D1 : public Base<D1> { void impl() { std::puts("D1::impl()"); } };
struct D2 : public Base<D2> { void impl() { std::puts("D2::impl()"); } };
 
#else // C++23 deducing-this syntax
 
struct Base { void name(this auto&& self) { self.impl(); } };
struct D1 : public Base { void impl() { std::puts("D1::impl()"); } };
struct D2 : public Base { void impl() { std::puts("D2::impl()"); } };
 
#endif
 
int main()
{
    D1 d1; d1.name();
    D2 d2; d2.name();
}

这里如果用多态实现，大致如下

C++
#include <cstdio>

struct Base
{
    virtual void name() { // 可按需设置为纯虚函数 }

};

struct D1 : public Base { void name() override { std::puts("D1::impl()"); } };
struct D2 : public Base { void name() override  { std::puts("D2::impl()"); } };

int main()
{
    D1 d1; d1.name();
    D2 d2; d2.name();
}

C++
template<typename T>
class Singleton {
public:
    static T& getInstance() {
        static T instance;
        return instance;
    }

    Singleton(const Singleton&) = delete;
    Singleton(Singleton&&) = delete;
    Singleton& operator=(const Singleton&) = delete;
    Singleton& operator=(Singleton&&) = delete;

protected:
    Singleton() = default;
    ~Singleton() = default;
};
